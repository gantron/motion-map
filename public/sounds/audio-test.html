// Audio Manager for MotionMap const audioManager = { sounds: {}, volume: 0.7, preload(soundName) { if (!this.sounds[soundName]) { const audio = new Audio(`/sounds/${soundName}.mp3`); audio.preload = 'auto'; audio.volume = this.volume; this.sounds[soundName] = audio; } return this.sounds[soundName]; }, play(soundName, loop = false) { const audio = this.preload(soundName); audio.loop = loop; audio.currentTime = 0; audio.play().catch(err => console.error('Audio error:', err)); }, stop(soundName) { if (this.sounds[soundName]) { this.sounds[soundName].pause(); this.sounds[soundName].currentTime = 0; } } }; // Usage examples: audioManager.play('click'); // Play click sound audioManager.play('ambient-loop', true); // Start ambient (looping) audioManager.stop('ambient-loop'); // Stop ambient
